def make_env(env_id: str, seed: int):
    gym = _import_gym()
    try:
        env = gym.make(env_id, render_mode="rgb_array")
    except Exception:
        env = gym.make(env_id)
    try:
        env.reset(seed=seed)
    except Exception:
        pass
    return GymEnvWrapper(env)
